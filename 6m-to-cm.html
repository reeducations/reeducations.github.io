[tex]\boxed{\tt{V_{rectangular \: prism} = length × width × height}}[/tex]

[tex]\tt{V_{rectangular \: prism} = 12 \: cm × 10 \: cm × 5 \: cm}[/tex]

[tex]\boxed{\tt{V_{rectangular \: prism} = 600 \: cm³}}[/tex]

2. [Length = 9 cm, Width = 6 cm, Height = ?, Volume = 540 cm]

[tex]\boxed{\tt{H = \frac{volume}{length × width}}}[/tex]

[tex]\tt{H = \frac{540 \: cm}{9 \: cm × 6 \: cm}}[/tex]

[tex]\tt{H = \frac{540 \: cm}{54 \: cm}}[/tex]

[tex]\boxed{\tt{H = 10 \: cm}}[/tex]
